\section{Some Proved Ideas}
\begin{thm}
Let $A$ and $B$ be two finite sets of points that are s-well-separated, let $x,p \in A$, and let $y,q \in B$. Then the following inequalities hold:

\begin{align*}
    |xy| &\leq (1 + 2/s) \cdot |xq| \\
    |xy| &\leq (1 + 4/s) \cdot |pq| \\
    |px| &\leq (2/s) \cdot |pq| \\
\end{align*}
\end{thm}

\begin{thm}
Within a bulb of radius $r$ the difference in shortest hamiltonian paths between any two points is at most $4r$. 
\end{thm}
\begin{proof}
    Fix one pair of endpoints $(a,b)$ and let $P_{a,b}$ be the optimal Hamiltonian path from $a$ to $b$ with length denoted $L(a,b)$.
    Next consider another pair of endpoints $(c,d)$ we can then build a hamiltonian path from $c$ to $d$ using $P_{a,b}$.
    \begin{enumerate}
        \item Start at $c$
        \item Travel from $c$ to $a$ for a cost of $d(c,a) \leq 2r$
        \item Follow the path $P_{a,b}$ from $a$ to $b$ for a cost of $c \leq L(a,b)$. (shortcutting $c,d$)
        \item Travel from $b$ to $d$ for a cost of $d(b,d) \leq 2r$
    \end{enumerate}
    Since the space obeys the triangle inequality we know that during step 3 we can only decrease the cost by shortcutting. Thus the total cost of this path is at most $L(a,b) + 4r$. Furthermore we know that this cost must ceil whatever the optimal hamiltonian path from $c$ to $d$ is. Thus we have shown that $L(c,d) \leq L(a,b) + 4r$. By symmetry of the argument we can also show that $L(a,b) \leq L(c,d) + 4r$. Combining these two inequalities we have that $\forall a,b,c,d;|L(a,b) - L(c,d)| \leq 4r$ as desired.
\end{proof}

Within a bulb of radius $r$ the difference in shortest hamiltonian paths between any two points is at most $2r$. 
\end{thm}

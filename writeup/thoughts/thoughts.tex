¥begin{defn}
    A ¥textbf{Pair Decomposition} of a point set $P$ is a set of pairs
    $$¥PairDecomp = ¥left¥{ ¥{A_1, B_1¥}, ¥ldots, ¥{A_k,B_k¥} ¥right¥}$$
    such that (I) $¥forall i; A_i,B_i ¥subset P$, (II) $¥forall i; A_i ¥cap B_i = ¥emptyset$, and (III) $¥bigcup_{i=1}^k A_i ¥otimes B_i = ¥binom{P}{2} = P ¥otimes P$.
    If $¥forall p,q ¥in P, p ¥neq q ¥implies ¥exists! i; ¥{p,q¥} ¥in A_i ¥otimes B_i$ then we say that $¥PairDecomp$ is a ¥textbf{Strict Pair Decomposition}.
¥end{defn}

¥begin{defn}
    $¥dist(Q,R) = ¥min_{q ¥in Q, r ¥in R} |qr|$
¥end{defn}

¥begin{defn}
    The pair $¥{Q,R¥}$ (where $Q ¥cap R = ¥emptyset$) is ¥textbf{$s$-well-separated} if
    $$¥max¥{¥diam(Q), ¥diam(R)¥} ¥leq ¥frac{1}{s} ¥cdot ¥dist(Q,R)$$
¥end{defn}

¥begin{corollary}
    Given two sets of points $A$ and $B$, there exists an $s$ such that $A$ and $B$ are $s$-well-separated if and only if $¥dist(A,B) > 0$.
¥end{corollary}

¥begin{defn}
    Let $P = A ¥cup B$ (with $A ¥cap B = ¥emptyset$). We say that $A$ and $B$ are ¥textbf{TSP-Separated} if there exists an optimal TSP tour on $P$ that visits all points in $A$ before any point in $B$ (or vice versa). 
    Furthermore, Let $P = A ¥cup B ¥cup C$ (with all three pairwise disjoint). We say that $A$, $B$, and $C$ are ¥textbf{TSP-Separated} if 
    there exists an optimal TSP tour on $P$ that has exactly 3 or 4 edges connecting the three sets. 
¥end{defn}

¥begin{defn}
    Let $¥{(A_1, B_1), ¥dots, (A_n, B_n)¥} = ¥PairDecomp$ be a Well Separated Pair Decomposition of the point set $P$.
    We say that a given pair $(A_i, B_i) ¥in ¥PairDecomp$ is a ¥textbf{$k$-binary WSP} if either $A_i ¥cup B_i = P$ (which we call $1$-binary WSP) or there exists another pair $(A_j, B_j) ¥in ¥PairDecomp$ such that $A_i ¥cup B_i$ is equal to either $A_j$ or $B_j$ and furthermore the pair $(A_j, B_j)$ is a $(k-1)$-binary WSP.
¥end{defn}

¥begin{defn}
    A Well Separated Pair Decomposition $¥PairDecomp$ of the point set $P$ is a ¥textbf{$k$-binary WSPD} if the union of all $k$-binary WSPs in $¥PairDecomp$ is equal to $P$.
¥end{defn}

¥begin{defn}
    A Well Separated Pair Decomposition $¥PairDecomp$ of the point set $P$ is a ¥textbf{full binary WSPD} if $¥forall (A_i, B_i) ¥in ¥PairDecomp$ the pair is a $k$-binary WSP for some $k$.
¥end{defn}

¥begin{thm}[The Fundamental Theorem of Well Separated Pairs]¥label{thm:fundamental_thm_wsp}
Let $A$ and $B$ be two sets of points that are $s$-well-separated, let $p,p' ¥in A$, and let $q,q' ¥in B$. 
Then the following inequalities hold:
¥begin{align*}
    |pq| &¥leq (1 + 1/s) ¥cdot |pq'| ¥¥
    |pq| &¥leq (1 + 2/s) ¥cdot |p'q'| ¥¥
    |pp'| &¥leq (1/s) ¥cdot |pq| ¥¥
¥end{align*}
¥end{thm}

¥begin{proof}
    Let $A$ and $B$ be two sets of points that are $s$-well-separated, let $p,p' ¥in A$, and let $q,q' ¥in B$.

    Note that $|qq'| ¥leq ¥diam(B) ¥leq ¥frac{1}{s} ¥cdot ¥dist(A,B) ¥leq ¥frac{1}{s} |pq'|$ by the definition of well-separated pairs.
    The first result then follows from the next analysis:
    ¥begin{align*}
        |pq| &¥leq |pq'| + |q'q| ¥¥
        &¥leq |pq'| + ¥frac{1}{s} |pq'| ¥¥
        &= (1 + 1/s) ¥cdot |pq'|
    ¥end{align*}

    For the second result we note that $|pp'| ¥leq ¥diam(A) ¥leq ¥frac{1}{s} ¥cdot ¥dist(A,B) ¥leq ¥frac{1}{s} |p'q'|$ by the definition of well-separated pairs.
    The second result then follows from the next analysis:
    ¥begin{align*}
        |pq| &¥leq |pp'| + |p'q'| + |q'q| ¥¥
        &¥leq |pp'| + |p'q'| + ¥frac{1}{s} |p'q'| ¥¥
        &¥leq ¥frac{1}{s} |p'q'| + |p'q'| + ¥frac{1}{s} |p'q'| ¥¥
        &= (1 + 2/s) ¥cdot |p'q'|
    ¥end{align*}
    
    Finally for the third result we note that the following is true $|pp'| ¥leq ¥diam(A) ¥leq ¥frac{1}{s} ¥cdot ¥dist(A,B) ¥leq ¥frac{1}{s} |pq|$ by the definition of well-separated pairs. The result is immediately aparent that $|pp'| ¥leq (1/s) ¥cdot |pq|$.
¥end{proof}

¥begin{lem}¥label[lem]{lem:wsp_subset}
    Let $A$ and $B$ be two sets of points that are $s$-well-separated.
    Then $A' ¥subseteq A$ and $B' ¥subseteq B$ (where $|A'|, |B'| ¥geq 1$) are also $s$-well-separated.
¥end{lem}
¥begin{proof}
    Let $A$ and $B$ be two sets of points that are $s$-well-separated and let $A' ¥subseteq A$ and $B' ¥subseteq B$ (where $|A'|, |B'| ¥geq 1$).
    We can then note that $¥dist(A', B') ¥geq ¥dist(A,B)$ due to the triangle inequality.
    We can also note that $¥diam(A') ¥leq ¥diam(A)$ and $¥diam(B') ¥leq ¥diam(B)$ since $A' ¥subseteq A$ and $B' ¥subseteq B$.
    Thus we have the following:
    $$¥max¥{¥diam(A'), ¥diam(B')¥} ¥leq ¥max¥{¥diam(A), ¥diam(B)¥} ¥leq ¥frac{1}{s} ¥cdot ¥dist(A,B) ¥leq ¥frac{1}{s} ¥cdot ¥dist(A', B')$$
    as desired.
¥end{proof}

¥begin{thm}
    Given a set of points $P$ within a ball of diameter $d$ within a metric space and starting point $x,y ¥in P$. Let $L$ be the length of some optimal Hamiltonian path $H$ between $x$ and $y$ over $P$. 
    Then the length of the optimal Hamiltonian path after removing $k$ intermediate points is at least $L - d(k+b)$ where $b$ is the number of non-contiguous blocks of points removed (under $H$).
¥end{thm}
¥begin{proof}
    Let $Q ¥subseteq P ¥setminus ¥{x,y¥}$ be the set of $k$ intermediate points that we remove from $P$.
    We then partition $Q$ into $b$ blocks of consecutive vertices such that $B_j = ¥{v_{j,1}, ¥dots, v_{j,m_j}¥}$ is the $j$-th block of points in $Q$ and furthermore there are no edges in $H$ between any two blocks of points in $Q$. 
    Thus we will have $¥sum_j m_j = k$.

    We define $H'$ to be an optimal Hamiltonian path between $x$ and $y$ over $P ¥setminus Q$ and to have length $L'$.

    We can then build a new Hamiltonian path $¥widehat H$ by inserting the blocks of points in $Q$ back into $H'$ in their same order as they appear in $H$.
    For a block $B_j$ we select any edge $(u_1, u_2)$ in the current version of $¥widehat H$ and replace it with 
    $(u_1, v_{j,1}), (v_{j,1}, v_{j,2}), ¥dots, (v_{j,m_j}, u_2)$.
    The increase in length will be
    $$¥dist(u_1, v_{j,1}) + ¥sum_{i=1}^{m_j-1} ¥dist(v_{j,i}, v_{j,i+1}) + ¥dist(v_{j,m_j}, u_2) - ¥dist(u_1, u_2)$$
    
    Since all distances are at most $d$, the increase is at most $(m_j + 1)d$.
    Then summing over all $b$ blocks we have the total increase at most $¥sum_{j=1}^b (m_j + 1)d = d(k+b)$.

    Then since $H$ is optimal we have that $L ¥leq L' + d(k+b)$ which implies that $L' ¥geq L - d(k+b)$ as desired.
¥end{proof}

¥begin{thm}
    Suppose sets $A$ and $B$ are $s$-well-separated for $s ¥geq ¥frac{3}{2}$.
    Consider a multiple optimal Hamiltonian paths problem of $k ¥geq 2$ paths over $2k$ endpoints, organized into $k$ pairs of endpoints.
    The objective is to minimize the total length of the $k$ paths, subject to the constraint that alternating between the paths and the predefined pairs forms exactly one continuous cycle.
    Suppose that for each predefined pair of endpoints, the two endpoints are always in the same set (either $A$ or $B$).
    Then there exists an optimal solution where across the entire solution only two edges cross between $A$ and $B$.
    Furthermore, if $s > ¥frac{3}{2}$ then all optimal solutions must satisfy this property.
¥end{thm}
¥begin{proof}
    To prove this theorem we will first show that the optimal solution will place all the $A$ endpoints consecutively in the cycle order and all the $B$ endpoints consecutively in the cycle order.
    Then we will leverage the previous theorems to show that under this required ordering an optimal solution exists which only has two edges crossing between $A$ and $B$.

    To begin, consider some optimal solution to the problem.

    ¥textbf{Case 1:} All endpoints are in the same set.
    Since the solution is optimal it is also an optimal solution to the multiple optimal Hamiltonian paths problem where the endpoints of each path are fixed as they are in the current solution.
    If $s > ¥frac{3}{2}$ then the current solution must have exactly two edges crossing between $A$ and $B$ by ¥Cref{thm:wsp_mpath_fixed_all_a}.
    If $s = ¥frac{3}{2}$ it follows that for the given configuration of endpoints there exists an optimal solution where only two edges cross between $A$ and $B$ by ¥Cref{thm:wsp_mpath_fixed_all_a}.

    ¥textbf{Case 2:} There are some endpoints in $A$ and some endpoints in $B$.
    Define the order of the endpoints in the cycle as $e_1, ¥dots, e_{2k}$ and let $P_1, ¥dots, P_k$ be the paths such that $P_i$ connects $e_{2i-1}$ to $e_{2i}$.
    Suppose that two endpoints $e_j, e_{j+1}$ (where $e_{2k+1} = e_1$) connected via some path $P_i$ in the solution are such that one is in $A$ and the other is in $B$.
    All alternations between sets must happen via paths since the predefined pairs of endpoints are all in the same set.
    Furthermore, after completing the cycle we must return to the same set we started in, thus there must be an even number of such pairs.
    Since there is at least one predefined pair of endpoints in both sets there must be at least two paths where one endpoint is in $A$ and the other is in $B$.

    Assume by contradiction that there are strictly more than two paths in the solution where one endpoint is in $A$ and the other is in $B$.
    Because the total number of such paths must be even, there must be at least 4 such paths.
    Let us consider any sequence of four such paths in the order they appear in the cycle.
    Because the predefined pairs never cross between sets, any sequence of crossing paths in cycle order must strictly alternate directions.
    Therefore, WLOG we can label the endpoints of these four paths as follows, the first path connects $a_1$ to $b_1$, the second path connects $b_2$ to $a_2$, the third path connects $a_3$ to $b_3$, and the fourth path connects $b_4$ to $a_4$ where $a_i ¥in A$ and $b_i ¥in B$ for all $i$.

    Note that on the path that goes from $a_1$ to $b_1$ there must be an edge directly connecting from $A$ to $B$ and we will label this edge as $¥{a^*, b^*¥}$ and from $a_3$ to $b_3$ the same is true and we label this edge as $¥{a^¥dagger, b^¥dagger¥}$.
    Then we can note the cycle goes:
    $$¥dots ¥sim ¥boxed{a_1 ¥leadsto a^* ¥to b^* ¥leadsto b_1} ¥sim ¥boxed{b_2 ¥leadsto a_2} ¥sim ¥boxed{a_3 ¥leadsto  a^¥dagger ¥to b^¥dagger ¥leadsto b_3} ¥sim ¥boxed{b_4 ¥leadsto a_4} ¥sim ¥dots.$$
    It follows that we can break apart the solution at the two edges $¥{a^*, b^*¥}$ and $¥{a^¥dagger, b^¥dagger¥}$ and reverse the order of the cycle and paths along the cutout section between these two edges to get a new solution where the cycle goes:
    $$¥dots ¥sim ¥boxed{a_1 ¥leadsto a^* ¥to a^¥dagger ¥leadsto a_3} ¥sim ¥boxed{a_2 ¥leadsto b_2} ¥sim ¥boxed{b_1 ¥leadsto b^* ¥to b^¥dagger ¥leadsto b_3} ¥sim ¥boxed{b_4 ¥leadsto a_4} ¥sim ¥dots.$$
    Because the predefined pairs are simply undirected pairings of endpoints within the exact same set, traversing them in reverse during this cycle modification is entirely valid and maintains all predefined pairings and does not create multiple cycles.

    Since the previous solution was assumed optimal we know that $|a^* b^*| + |a^¥dagger b^¥dagger| ¥leq |a^* a^¥dagger| + |b^* b^¥dagger|$.
    Since $A$ and $B$ are $s$-well-separated we know that $¥forall i,j,k; |a_i a_j| ¥leq ¥frac{1}{s} |a_i b_k|$ (and vice versa for $B$).
    From this we can conclude:
    ¥begin{align*}
        |a^* b^*| + |a^¥dagger b^¥dagger| &¥leq |a^* a^¥dagger| + |b^* b^¥dagger| ¥¥
        &¥leq ¥frac{1}{s} |a^* b^*| + ¥frac{1}{s} |a^¥dagger b^¥dagger| ¥¥
        &¥leq ¥frac{1}{s} (|a^* b^*| + |a^¥dagger b^¥dagger|)
    ¥end{align*}
    This implies that $1 ¥leq ¥frac{1}{s}$ which is a contradiction since $s ¥geq ¥frac{3}{2}$.
    Therefore there must be exactly two paths where one endpoint is in $A$ and the other is in $B$ as desired.

    Since the solution is optimal it is also an optimal solution to the multiple optimal Hamiltonian paths problem where the endpoints of each path are fixed as they are in the current solution.
    If $s > ¥frac{3}{2}$ then the current solution must have exactly two edges crossing between $A$ and $B$ by ¥Cref{thm:wsp_mpath_fixed_mixed}.
    If $s = ¥frac{3}{2}$ it follows that for the given configuration of endpoints there exists an optimal solution where only two edges cross between $A$ and $B$ by ¥Cref{thm:wsp_mpath_fixed_mixed}.

    Therefore in both cases we have shown that there exists an optimal solution where across the entire solution only two edges cross between $A$ and $B$ as desired and furthermore, if $s > ¥frac{3}{2}$ then all optimal solutions must satisfy this property as desired.
¥end{proof}

¥begin{thm}
    Suppose sets $A$ and $B$ are $s$-well-separated for $s ¥geq ¥frac{3}{2}$.
    Consider a multiple optimal Hamiltonian paths problem of $k ¥geq 2$ paths over $2k$ endpoints, organized into $k$ pairs of endpoints.
    The objective is to minimize the total length of the $k$ paths, subject to the constraint that alternating between the paths and the predefined pairs forms exactly one continuous cycle.
    Suppose that there is a non-zero ¥textbf{even} number of predefined pairs with one point in $A$ and the other in $B$.
    Then there exists an optimal solution where no path has an edge connecting $A$ to $B$.
    Furthermore, if $s > ¥frac{3}{2}$ then all optimal solutions must satisfy this property.
¥end{thm}
¥begin{proof}
    To show this we first show that in the optimal solution the endpoint cycle is ordered such that no path has its endpoints in different sets.
    Then our previous theorems will imply that given the required ordering of the endpoints in the cycle, there exists an optimal solution where no path has an edge connecting $A$ to $B$.

    To begin, consider some optimal solution to the problem.
    By ¥Cref{thm:wsp_mpath_fixed_mixed}, there must exist a solution such that if a given path in the solution starts and ends in the same set then it has no edges connecting $A$ to $B$ and if a given path in the solution starts and ends in different sets then it has exactly one edge connecting $A$ to $B$.
    If $s > ¥frac{3}{2}$ then the current solution must satisfy this property by ¥Cref{thm:wsp_mpath_fixed_mixed} and if $s = ¥frac{3}{2}$ then WLOG we assume the current solution satisfies this property.

    Assume that there is a path $P_i$ in the solution that has endpoints in different sets and thus has an edge connecting $A$ to $B$.
    WLOG assume that $P_i$ connects $a' ¥in A$ to $b' ¥in B$.
    Let $C$ denote the continuous cycle formed by alternating between the endpoints of the paths and the predefined pairs in the solution.
    Let us establish an arbitrary traversal direction along $C$, against which the direction of all paths will be relative to.
    Since $C$ is closed we know that there must be an even number of alternations between $A$ and $B$ along the cycle.
    Because there is an even number of predefined pairs with one point in $A$ and the other in $B$ then the number of paths with one endpoint in $A$ and the other in $B$ must also be even to ensure that the total number of alternations between $A$ and $B$ along the cycle is even.
    Thus there must be another path $P_j$ in the solution that has an edge connecting $A$ to $B$.

    ¥textbf{Case 1:} $P_i$ and $P_j$ are in the same direction.
    Let $P_j$ connect $a'' ¥in A$ to $b'' ¥in B$.
    We can then note that in both $P_i$ and $P_j$ there must be an edge directly connecting from $A$ to $B$ and we will label these edges as $¥{a^*, b^*¥}$ and $¥{a^¥dagger, b^¥dagger¥}$ respectively.
    Then we can note the cycle goes:
    $$¥dots ¥sim ¥boxed{a' ¥leadsto a^* ¥to b^* ¥leadsto b'} ¥sim ¥dots ¥sim ¥boxed{a'' ¥leadsto a^¥dagger ¥to b^¥dagger ¥leadsto b''} ¥sim ¥dots.$$
    It follows that we can break apart the solution at the two edges $¥{a^*, b^*¥}$ and $¥{a^¥dagger, b^¥dagger¥}$ and reverse the order of the cycle and paths along the cutout section between these two edges to get a new solution where the cycle goes:
    $$¥dots ¥sim ¥boxed{a' ¥leadsto a^* ¥to a^¥dagger ¥leadsto a''} ¥sim ¥dots ¥sim ¥boxed{b' ¥leadsto b^* ¥to b^¥dagger ¥leadsto b''} ¥sim ¥dots.$$
    Because the predefined pairs are simply undirected pairings of endpoints, traversing them in reverse during this cycle modification is entirely valid and maintains all predefined pairings and does not create multiple cycles.

    Since the previous solution was assumed optimal we know that $|a^* b^*| + |a^¥dagger b^¥dagger| ¥leq |a^* a^¥dagger| + |b^* b^¥dagger|$.
    Since $A$ and $B$ are $s$-well-separated we know that $¥forall i,j,k; |a_i a_j| ¥leq ¥frac{1}{s} |a_i b_k|$ (and vice versa for $B$).
    From this we can conclude:
    ¥begin{align*}
        |a^* b^*| + |a^¥dagger b^¥dagger| &¥leq |a^* a^¥dagger| + |b^* b^¥dagger| ¥¥
        &¥leq ¥frac{1}{s} |a^* b^*| + ¥frac{1}{s} |a^¥dagger b^¥dagger| ¥¥
        &¥leq ¥frac{1}{s} (|a^* b^*| + |a^¥dagger b^¥dagger|)
    ¥end{align*}
    This implies that $1 ¥leq ¥frac{1}{s}$ which is a contradiction since $s ¥geq ¥frac{3}{2}$.
    Therefore there cannot be two paths in the same direction that have edges connecting $A$ to $B$.

    ¥textbf{Case 2:} $P_i$ and $P_j$ are in different directions.
    Let $P_j$ connect $b'' ¥in B$ to $a'' ¥in A$.
    By Case 1 it follows that these must be the only two paths crossing between $A$ and $B$ in the solution since if there were any more we could find two in the same direction.

    We can then note that in $P_i$ there must be an edge directly connecting from $A$ to $B$ and we will label this edge as $¥{a^*, b^*¥}$ and in $P_j$ there must be an edge directly connecting from $B$ to $A$ and we will label this edge as $¥{b^¥dagger, a^¥dagger¥}$.
    By linking $a^*$ to $a^¥dagger$ and $b^*$ to $b^¥dagger$ we can break apart the solution at the two edges $¥{a^*, b^*¥}$ and $¥{b^¥dagger, a^¥dagger¥}$ and create two disjoint cycles based on the structure of the original cycle.
    Let $C_A$ be the sub-cycle that contains $a^*$ and $a^¥dagger$ (and consequently $a'$ and $a''$) and let $C_B$ be the sub-cycle that contains $b^*$ and $b^¥dagger$ (and consequently $b'$ and $b''$).
    Since all alternations must be done via predefined pairs, at least one of these two sub-cycles must contain a predefined pair of endpoints with one endpoint in $A$ and the other in $B$.
    WLOG assume that $C_A$ contains such a predefined pair of endpoints (otherwise $C_B$ would contain it and the definition of $A$ and $B$ could be flipped).
    
    Let $(a_1, b_1)$ be the first predefined pair of endpoints in $C_A$ with $a_1 ¥in A$ and $b_1 ¥in B$ after $a^*$ in the cycle order of $C_A$.
    Let $b_2 ¥in B$ be the endpoint on the other end of the path that begins with $b_1$ (since $b'$ and $b''$ are strictly contained within $C_B$, the endpoint $b_1 ¥in C_A$ must belong to a path other than $P_i$ and $P_j$ which represent the only two paths with endpoints in different sets).
    We can choose any edge from the $b_1$ to $b_2$ path and it will always connect two points in $B$ by the previous application of ¥Cref{thm:wsp_mpath_fixed_mixed}.
    Considering the first edge in the $b_1$ to $b_2$ path, it follows that the first edge must connect $b_1$ to some point in $B$ and we will label this point $b_3$.

    We can then break apart the solution at the edge $¥{b_1, b_3¥}$ and create a new solution by adding $¥{b_1, b^*¥}$, $¥{b^¥dagger, b_3¥}$, and $¥{a^*, a^¥dagger¥}$ all back in.
    This new solution maintains all predefined pairs and forms a single cycle.
    To rigorously verify this, note that removing $¥{a^*, b^*¥}$, $¥{b^¥dagger, a^¥dagger¥}$, and $¥{b_1, b_3¥}$ breaks the original cycle into three disjoint directed segments: $S_1$ from $b_3 ¥leadsto a^*$, $S_2$ from $b^* ¥leadsto b^¥dagger$, and $S_3$ from $a^¥dagger ¥leadsto b_1$.
    Replacing the cut edges with $¥{b_1, b^*¥}$, $¥{b^¥dagger, b_3¥}$, and $¥{a^*, a^¥dagger¥}$ recombines these segments into the sequence:
    $$S_1 ¥xrightarrow{¥{a^*, a^¥dagger¥}} S_3 ¥xrightarrow{¥{b_1, b^*¥}} S_2 ¥xrightarrow{¥{b^¥dagger, b_3¥}} S_1.$$
    Because the new topology visits all three segments exactly once before returning to the start, it forms a single continuous cycle.
    Furthermore, since none of $S_1$, $S_2$, and $S_3$ are created by cutting through any predefined pair of endpoints, the new solution maintains all predefined pairs of endpoints.

    Since the previous solution was assumed optimal we know that $|a^* b^*| + |b^¥dagger a^¥dagger| + |b_1 b_3| ¥leq |a^* a^¥dagger| + |b_1 b^*| + |b^¥dagger b_3|$.
    Define $¥omega = ¥dist(A,B)$.
    The following are true by the definition of well-separated pairs and non-negativity of distances:
    $$
        |a^* b^*| ¥geq ¥omega ¥quad
        |b^¥dagger a^¥dagger| ¥geq ¥omega ¥quad
        |b_1 b_3| ¥geq 0 ¥quad
        |a^* a^¥dagger| ¥leq ¥frac{1}{s} ¥omega ¥quad
        |b_1 b^*| ¥leq ¥frac{1}{s} ¥omega ¥quad
        |b^¥dagger b_3| ¥leq ¥frac{1}{s} ¥omega.
    $$
    From these we can conclude: $2¥omega ¥leq |a^* b^*| + |b^¥dagger a^¥dagger| + |b_1 b_3| ¥leq |a^* a^¥dagger| + |b_1 b^*| + |b^¥dagger b_3| ¥leq ¥frac{3}{s} ¥omega$ which implies that $s ¥leq ¥frac{3}{2}$.
    If $s > ¥frac{3}{2}$ then this is a contradiction and thus the original solution must not have had two paths with endpoints in different sets.
    It immediately follows from this and the previous application of ¥Cref{thm:wsp_mpath_fixed_mixed} that there are no edges connecting $A$ to $B$ in the current solution as desired.
    If $s = ¥frac{3}{2}$ then we can note that all these inequalities must be tight and thus since the only two edges crossing between $A$ and $B$ were removed from the original solution to create the new solution, it follows that the new solution must also be optimal and has no paths with edges connecting $A$ to $B$.
¥end{proof}

    If $s > 1$ then all optimal Hamiltonian paths must satisfy this property.
    % TODO: what if we got A to B to A then to C to B to C to A? we would have four paths connecting A to B
¥end{thm}
¥begin{proof}
    Consider some optimal Hamiltonian path over $P$ which starts at $a' ¥in A_i$ and ends at $a'' ¥in A_j$.
    Assume there is two pairs of sets $B = A_k$ and $C = A_l$ which are directly connected by two edges in the same direction (where $k ¥neq l$).
    WLOG let these edges be $(b_1, c_1)$ and $(b_2, c_2)$ where $b_1, b_2 ¥in B$ and $c_1, c_2 ¥in C$ ordered as they appear in the path (if $C$ occurs before $B$ then flip the names).

    This gives the following partial structure of the path:
    $$a' ¥leadsto b_1 ¥to c_1 ¥leadsto b_2 ¥to c_2 ¥leadsto a''$$
    Since the path is optimal we know that it satisfies the 2-optimality condition implying that $|b_1 c_1| + |b_2 c_2| ¥leq |b_1 b_2| + |c_1 c_2|$.
    This is to say that the following path must be at least as long as the original path:
    $$a' ¥leadsto b_1 ¥to b_2 ¥leadsto c_1 ¥to c_2 ¥leadsto a''$$
    Since $C$ and $B$ are $s$-well-separated we know that $¥forall i,j,k; |b_i b_j| ¥leq ¥frac{1}{s} |b_i c_k|$ (and vice versa for $C$). 
    From this we can conclude the following:
    ¥begin{align*}
        |b_1 c_1| + |b_2 c_2| &¥leq |b_1 b_2| + |c_1 c_2| ¥¥
        &¥leq ¥frac{1}{s} |b_1 c_1| + ¥frac{1}{s} |b_2 c_2| ¥¥
        &= ¥frac{1}{s} (|b_1 c_1| + |b_2 c_2|)
    ¥end{align*}
    This implies that $1 ¥leq ¥frac{1}{s}$ which in turn implies that $s ¥leq 1$.
    Thus if $s > 1$ then we have a contradiction and therefore no optimal Hamiltonian path can have more than two edges directly connecting any pair of sets in the same direction.
    If $s = 1$ then we can conclude that $|b_1 c_1| + |b_2 c_2| = |b_1 b_2| + |c_1 c_2|$ and therefore the second path is also an optimal Hamiltonian path over $P$.
    We can repeat this process over all pairs until every set is connected by no more than one edge to another set in a given direction.

    Therefore there exists an optimal Hamiltonian path over $P$ which starts at $a' ¥in A_i$ and ends at $a'' ¥in A_j$ where no set is ever directly connected to another set more than twice in the same direction.
¥end{proof}

¥begin{algorithm}[H]
¥caption{¥textsc{BalancedMetricSplit2}$(D,s,k)$}
¥label{alg:balanced_metric_split2}
¥begin{algorithmic}[1]
¥Require A metric distance matrix $D ¥in ¥mathbb{R}^{n¥times n}$ (symmetric, triangle inequality), separation factor $s>0$, and number of clusters $k ¥ge 2$
¥Ensure A no more than $k$-partition $¥mathcal{P}=¥{S_1,¥dots,S_k¥}$ that maximizes balance such that every pair of clusters is $s$-well-separated (or a single cluster of all points is returned)

¥State $¥mathcal{M} ¥gets ¥{D_{ij} ¥mid 1 ¥le i < j ¥le n¥}$ ¥Comment{candidate diameter thresholds}
¥State $¥mathcal{S}^¥star ¥gets ¥big¥{ ¥{1,2,¥dots,n¥} ¥big¥}$; $¥text{bestScore} ¥gets n$

¥ForAll{$M ¥in ¥mathcal{M}$}
    ¥State $E_M ¥gets ¥{(i,j)¥mid 1¥le i<j¥le n,¥; D_{ij} < s¥cdot M¥}$
    ¥State $G_M ¥gets (V=¥{1,¥dots,n¥}, E_M)$
    ¥State $¥mathcal{C} ¥gets ¥textsc{ConnectedComponents}(G_M)$

    ¥If{$1 < |¥mathcal{C}| ¥leq l$}
        ¥State ¥textbf{continue}
    ¥EndIf

    ¥Comment{Diameter feasibility: ensure each component has diameter $¥le M$}
    ¥State $¥text{ok} ¥gets ¥textbf{true}$
    ¥ForAll{$C ¥in ¥mathcal{C}$}
        ¥If{$¥diam(C) > M$}
            ¥State $¥text{ok} ¥gets ¥textbf{false}$
            ¥State ¥textbf{break}
        ¥EndIf
    ¥EndFor
    ¥If{¥textbf{not} ok}
        ¥State ¥textbf{continue}
    ¥EndIf

    ¥Comment{Separation feasibility: ensure pairwise separation}
    ¥State $¥text{sep} ¥gets ¥textbf{true}$
    ¥ForAll{distinct $C, C' ¥in ¥mathcal{C}$}
        ¥State $¥Delta(C,C') ¥gets ¥min¥{D_{xy}¥mid x¥in C,¥; y¥in C'¥}$
        ¥If{$¥Delta(C,C') < s ¥cdot ¥max(¥diam(C),¥diam(C'))$}
            ¥State $¥text{sep} ¥gets ¥textbf{false}$
            ¥State ¥textbf{break}
        ¥EndIf
    ¥EndFor
    ¥If{¥textbf{not} sep}
        ¥State ¥textbf{continue}
    ¥EndIf

    ¥Comment{Balance objective: minimize size of largest cluster}
    ¥State $¥text{score} ¥gets ¥max_{C¥in¥mathcal{C}} ¥bigl|C¥bigr|$
    ¥If{$¥text{score} < ¥text{bestScore}$}
        ¥State $¥text{bestScore} ¥gets ¥text{score}$
        ¥State $¥mathcal{S}^¥star ¥gets ¥mathcal{C}$
    ¥EndIf
¥EndFor

¥State ¥Return $¥mathcal{S}^¥star$
¥end{algorithmic}
